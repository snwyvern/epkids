<% def getSubTable= { p->
	if( p.id != 'xf_used_default_buttons'){
		 return ""
	}
	def json = new groovy.json.JsonSlurper()
	def buttons = json.parseText(p.value);
	def t= '\n[cols="3*", options="header"]\n'
	t += '!===\n'
	t += '!Label!Icon!Action\n'
	buttons.items.each{
		t += '!'+it.label
		t += '!'+it.icon
		t += '!'+it.action+'\n'
		t += '\n'
	}
	t +='!===\n'
	return t
}%>
