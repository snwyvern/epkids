    t.onTap = function( e ) {
      console.log( "Tap:", t.result );
      //sendTestRequest( e.detail );

  } );

  function sendTestRequest( data ) {
    var params = {
      service: "simpl4",
      method: simpl4.util.BaseManager.getNamespace() + ".sendTestRequest",
      parameter: data,
      async: true,
      context: this,
      failed: function( e ) {
        console.error( "sendTestRequest:", e );
      },
      completed: function( ret ) {
        console.log( "sendTestRequest:", ret );
      }
    }
    simpl4.util.Rpc.rpcAsync( params );
  }
</script>
